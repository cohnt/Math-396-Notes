\par\noindent Recall: $\L^{k}(V)=\set{f:V^{k}\to\R\;|\;f\ptxt{ multilinear}}$.\n
$f\in\L^{k}(V)$ and $g\in\L^{l}(V)$ yields $f\ot{}g\in\L^{k+l}(V)$.\n
Rules:
\begin{itemize}[label=-,itemsep=0pt,topsep=0pt]
	\item $f\ot{}g$ is linear w.r.t $f$ and $g$
	\item $f\ot(g\ot{}h)=(f\ot{}g)\ot{}h$
	\item $T^{*}(f\ot{}g)=T^{*}f\ot{}T^{*}g$
	\item For $I=(i_{1},\ldots,i_{k})$, $\phi_{I}=\phi_{i_{1}}\ot\cdots\ot\phi_{i_{k}}$
\end{itemize}

\par\noindent\n
$\A^{k}(V)=\set{f\in\L^{k}(V)\;|\;f\ptxt{ alternating}}$.\n

\par\noindent Given $f\in\A^{k}(V),g\in\A^{l}(V)$, we don't necessarily have $f\ot{}g\in\A^{k+l}(V)$.\n

\thm{There is some map $\wedge:\map{\A^{k}\times\A^{k}}{\A^{k+l}}{(f,g)}{f\wedge{}g}$ which satisfies
\begin{enumerate}[label=(\alph*),itemsep=0pt,leftmargin=3.5\parindent]
	\item $f\wedge{}g$ is linear in $f$ and linear in $g$
	\item $(f\wedge{}g)\wedge{}h=f\wedge(g\wedge{}h)$
	\item $g\wedge{}f=(-1)^{kl}f\wedge{}g$
	\item $\psi_{I}=\psi_{i_{1}}\wedge\cdots\wedge\psi_{i_{k}}$
	\item $T^{*}(f\wedge{}g)=T^{*}f\wedge{}T^{*}g$
\end{enumerate}}

\par\noindent From last time, we have a basis for $\Alt^{k}(V)$ $\psi_{I}=\sum_{\sigma\in{}S_{k}}\sgn\sigma\cdot\psi_{I_{\sigma}}$ where $I_{\sigma}=(i_{\sigma(1)},\ldots,i_{\sigma(k)})$.\n
Note that $\psi_{i}=\phi_{i}$.\n

\par\noindent Rules determine the operations:
\begin{multicols}{2}
\centerline{$\displaystyle{}f=\smashoperator[lr]{\sum_{\scalebox{0.5}{$\begin{array}{c}I\asc\ptxt{$k$-tuple}\\ \ptxt{entries}\in\set{1,\ldots,n}\end{array}$}}}\alpha_{I}\psi_{I}$}\n
\centerline{$\displaystyle{}g=\smashoperator[lr]{\sum_{\scalebox{0.5}{$\begin{array}{c}J\asc\ptxt{$k$-tuple}\\ \ptxt{entries}\in\set{1,\ldots,n}\end{array}$}}}\beta_{J}\psi_{J}$}
\end{multicols}
\[
f\wedge{}g=\smashoperator[lr]{\sum_{\scalebox{0.75}{$\begin{array}{c}I\asc\\ J\asc\end{array}$}}}\alpha_{I}\beta_{J}\psi_{I}\wedge\psi_{J}=\bigsum{0.75}{\begin{array}{c}I\asc\\ J\asc\\ \ptxt{no duplicates}\\ I_{\asSet}\cap{}J_{\asSet}=\emptyset\end{array}}\alpha_{I}\beta_{J}\sgn(I,J)\psi_{\sort(I,J)}
\]

\par\noindent Where $\sgn(I,J)=(-1)^{\ptxt{\# transpositions to set }(I,J)}$.\n

\par\noindent Claim: $\wedge$ defined by this formula satisfies conditions (a) through (e).
