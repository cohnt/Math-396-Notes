\par\noindent Recall: $A^{\ptxt{osso}\C}\xrightarrow{f=u+iv}\C$. $f$ is \u{holomorphic} if and only if $f$ satisfies one/all of the following equivalent conditions:
\begin{enumerate}[label=\arabic*), topsep=0pt, itemsep=0pt]
	\item $f\in{}C^{1}$ and $f\,dz$ is closed.
	\item $u,v\in{}C^{1}(A,\R)$ satisfy the CR equations.
	\item $f$ is $\C$-differentiable at each point of $A$ and $f'_{\C}$ is continuous on $A$.
\end{enumerate}

\thm{If $f:A\to{}B$ is a holomorphic diffeomorphism, then $f\inv:B\to{}A$ is a holomorphism.\n
Proof: Write $f=\smallPMatrix{u\\ v}$, $f\inv=\smallPMatrix{\tilde{u}\\ \tilde{v}}$. $D\smallPMatrix{u\\ v}(z_{0})=\smallPMatrix{\alpha & -\beta\\ \beta & \alpha}$.\n
By Thm 7.4, $D\smallPMatrix{\tilde{u}\\ \tilde{v}}(f(z_{0}))=\smallPMatrix{\alpha & -\beta\\ \beta & \alpha}\inv=\smallPMatrix{\frac{\alpha}{\alpha^{2}+\beta^{2}} & \frac{\beta}{\alpha^{2}+\beta^{2}}\\ \frac{-\beta}{\alpha^{2}+\beta^{2}} & \frac{\alpha}{\alpha^{2}+\beta^{2}}}=\frac{1}{\alpha^{2}+\beta^{2}}\smallPMatrix{\alpha & \beta\\ -\beta & \alpha}$.\n
This satisfies the CR equations, so $f\inv$ is holomorphic.\proven}

\thm{(Cauchy Integral Theorem V.2) Given $M^{\ptxt{cpt}\;2\ptxt{-mfd}}\subset\C$, $f$ holomorphic on some neighborhood of $M$, then\n
\inlineeq{
	\int_{\partial{}M}f\,dz\overset{\ptxt{Stokes'}}{=}\int_{M}d(f\,dz)=\int_{M}0=0
}}

\par\noindent Important special case: when $\partial{}M$ is the union of $2$ connected curves $C_{1}$ and $C_{2}$, we have
\[
0=\int_{\partial{}M}f\,dz=\int_{C_{1}}f\,dz-\int_{C_{2}}f\,dz
\]

\par\noindent So $\displaystyle\int_{C_{1}}f\,dz=\int_{C_{2}}f\,dz$.\n

\cor{Given $z_{0}\in{}U^{\ptxt{osso}\C}$, $f$ holomorphic on $U\cut\set{z_{0}}$, then $\displaystyle\smashoperator[r]{\int_{\set{z:\abs{z-z_{0}}=r}}}f\,dz$ is independent of $r\in(0,d(z_{0},\Bd{}U))$. (Note that this set is a circle; we're traveling counterclockwise.)}

\defn{This integral is equal to $2\pi{}i\Res(f\,dz,z_{0})$, where $\Res(f\,dz,z_{0})$ is the \u{residue} of $f\,dz$ at $z_{0}$.}

\cor{(Residue Theorem) Given $M^{\ptxt{cpt}\;2\ptxt{-mfd}}\subset{}U^{\ptxt{open}}\subset\C$, $z_{1},\ldots,z_{k}$ distinct in $M\cut\partial{}M$, and $f$ holomorphic on $U\cut\set{z_{1},\ldots,z_{k}}$, then\n
\inlineeq{
	\int_{\partial{}M}f\,dz=2\pi{}i\sum_{j=1}^{k}\Res(f\,dz,z_{j})
}
Proof: Draw small $\epsilon$-balls around each $z_{j}$ which don't intersect with $\partial{}M$. Then apply V.2 of the Cauchy Integral Theorem.\proven}

\par\noindent The \i{\b{most}} important special case: consider $g$ holomorphic on $U\ni{}z_{0}$, $f(z)=\frac{g(z)}{z-z_{0}}$. Then
\[
\smashoperator[lr]{\int_{\set{z:\abs{z-z_{0}}=r}}}f(z)\,dz=\int_{0}^{2\pi}\frac{g(z_{0}+re^{it})}{re^{it}}ire^{it}\,dt=i\int_{0}^{2\pi}g(z_{0}+re^{it})\,dt
\]

\par\noindent This is independent of $r$, so
\[
\lim_{r\to{}0}i\int_{0}^{2\pi}g(z_{0}+re^{it})\,dt=i\int_{0}^{2\pi}g(z_{0})\,dt=2\pi{}ig(z_{0})
\]

\par\noindent because the limit converges uniformly. Thus, $\Res\paren{\frac{g(z)}{z-z_{0}},z_{0}}=g(z_{0})$.\n

\cor{(Cauchy Integral Formula) Given $M^{\ptxt{cpt}\;2\ptxt{-mfd}}\subset\C$, $g$ holomorphic on $U^{\ptxt{osso}\;\C}\supset{}M$, then\n
\inlineeq{
	\int_{\partial{}M}\frac{g(z)}{z-z_{0}}\,dz=2\pi{}ig(z_{0})
}}

\par\noindent Unfortunately, this doesn't work in $\R$.
